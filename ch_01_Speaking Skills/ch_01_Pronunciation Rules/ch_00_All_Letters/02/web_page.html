A

يلفظ حرف (a) كحرف الياء إذا انتهت الكلمة بحرف (e) أمثلة:

تاريخ,زمن ـــــDateـــــ ديـت


يلفظ حرف (a) في الكلمات الثلاثية كالألف الممدودة في اللغة العربية وفي بعض الكلمات الرباعية والخماسية أمثلة:

بعيد ــــــــــــــــــFarـــــ فار

ذيل ــــــTailـــــ تيـل


إذا كان (a) متبوعاً بحرف (u) فأنه ينطق كحرف (o) أمثلة:

قاعةـــــــــHallـــــ هول

صغيرـــــSmallـــــ سمول


B

يخدر,يفقد الحس ـــــNumbــــــ نـم

متقزح اللون ـــــIridescentـــــ اردسنت

لا يلفظ الحرف(c) أيضا إذا جاء في أول الكلمة مثل:

مشهد,منظرــــــSceneـــــ سين


يلفظ حرف(c) كما هو في الحروف الهجائية (سي) إذا جاء بعده الاحرف(e) (i) (y) أمثلة:

قطعةــــــــــــــــــPieceــــــ بايس

دراجةــــــــــــــــــBicycleـــــ باي سِِكِل

قلم رصاص ـــــPencilـــــ بنسِل

قطة Cat

D

يوم الاربعاءـــــWednesdayــــــ ونزدي


E

إذا وجد الحرف (e) في الكلمة فأنه ينطق مابين الفتح والكسر أمثلة:

سريرــــــBedـــــ بـد

خيمةـــــــTentـــــ تِنت

شجرةــــــــTreeـــــ تري


إذا كان حرف(e) متبوعاً بحرف(a) فأنه يلفظ كحرف الياء أمثلة:

مقعدـــــــــــــــــــSeatـــــ سيت

لا يلفظ الحرف (e) في نهاية جميع الكلمات أمثلة:

شاطيءـــــShoreـــــــ شور

حياةـــــــــــــLifeـــــــ لايف

لا يلفظ هذا الحرف إذا سبقه الحرف(i) وتلاه الحرف(n) ووقع (n) في آخر الكلمة أمثلة:


علامة,رمزـSign ساين

لا يلفظ أيضاً إذا سبقه (u) وتلاه(n) مثال:

يناقش ــــــOppugnـــــ أبيون

لا يلفظ إذا جاء في أول الكلمة وتلاه(n) مثال:

يزعج ـــــgnawـــــ نوو


حرف (g) له نطقين احياناً ينطق كالجيم السعودية واحياناً كالجيم المصرية والقاعدة هي : إذا جاء بعده احد الاحرف (y) (i) (e) ينطق كالجيم السعودية . وإذا جاء بعده أي حرف غيرها وإذا جاء في آخر الكلمة كالجيم المصرية أمثلة:

مهندس ــــــEngineerــــــ انجنِير


H

لا يلفظ الحرف(h) إذا أتى مباشرة بعد الحرف(g) وتلاهما حرفاً صوتياً أمثلة:

شبح ــــــــGhostـــــ ﭼوست

البوذيةـــــBuddhismـــــ بودزم


لا يلفظ (h) إذا أتى بعد الحرف(c) وتلاهما حرف صوتي مثل:

انفصال ـــــSchismـــــ سزم

مدرسة ــــــــSchoolـــــ سكول


لايلفظ إذا أتى في أول الكلمة وتلاه الحرفان (o) (n) أمثلة:

احترام,شرف ـــــHonourـــــ أنور

قافيةـــــــــRhymeـــــ ريـم

ايقاع ـــــRhythmــــ رذم


لا ينطق الحرف (h) إذا جاء بعد الحرف(w) أمثلة:

أبيض ــــــــــــــــــــــــــــــWhiteـــــ وايت

ويحدث العكس إذا جاء (o) بعد(h) ينطق (h) ولا ينطق (w) أمثلة:

من ـــــــــــــــــWhoــــــ هُـو

بيع الجملةــــــWholesaleــــــ هول سَيـل


لا ينطق حرف(h) في نهاية جميع الكلمات مثل:

مروحةـــــPunkahـــــ بـانكْ


لا ينطق حرف(h) إذا جاء بعد حرف(e) وتلاهما حرفاً صوتياً مثل:

عنيف ـــــVehementـــــــ فِـيمُـنت

عربةـ Vehicle فِـي ايكل


I

إذا جاء حرف (i) في الكلمة يلفظ مثل الكسرة أمثلة:

في ــــــــــInـــــ أن

يضرب ـــــHitـــــ هِـت

اجلس ــــــــSitـــــ سِـت


إذا جاء حرف(i) في وسط الكلمة اوفي بدايتها وكانت منتهية بحرف(e) يلفظ كما هو في حروف الهجاء ( آي) أمثلة:

وقت ــــــــTimeـــــ تايـم

يسوق ــــــــDriveـــــ درايف

ثلج ـــــــــــــIceـــــ آيـس

يعيش,يسكن ـــــLiveـــــ لـيف

منذ ــــــــــــــــــSinceـــــ سيـنس


إذا جاء حرفا (gh) بعد حرف(i) فأنهما لا يلفظان ويلفظ حرف(i) ممدوداً أمثلة:

قريب من ـــــ Nighـــــ ناي

عالي ــــــــــــHighـــــ هاي


K

لا يلفظ هذا الحرف إذا جاء بعده (n) . يعني اعتبر الكلمة تبدأ (n) أمثلة:

سكين ــــKnifeـــــ نايف

عقدةـــــــــKnotــــــ نـوت


L

لا ينطق هذا الحرف إذا جاء قبل الاحرف (d)( f)(k) أمثلة:

يمكن أن ـــــCouldــــــ كـود

كلام ـــــــــــــــTalkــــــــ تـوك

ذئب ـــــWolfــــ تنطق ولف. والعالم ـــــWorldــــ تنطق ورلد. والذهب ـــــGoldــــ تنطق ﭼولد

N

يرسم ــــــLimnـــــ لِـم

O

شكل ـــــFormـــــ فورم

قلعة ـــــــــــFortــــــ فورت

T

لا يلفظ في بعض الكلمات إذا جاء بعد (f ) ووقع قبلهما الحرف (o) وبعدهما الحرف (e) أمثلة:

مملوءـــــFullـــــ فـل
